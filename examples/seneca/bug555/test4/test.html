Test for bug 555

Test for drawing opaque spheres behind and in front of transparent spheres.
Colors are slightly off, which is known. 

// Test 
import processing.opengl.*;

void setup() {
  size(100, 100, P3D);
  background(33, 66, 99);
  noStroke();

  // opaque first behind
  // transparent second front
  pushMatrix();
  translate(width/4, height/4, 0);
  fill(255);
  sphere(20);
  popMatrix();
  
  pushMatrix();
  translate(width/4, height/4, 30);
  fill(255, 0, 0, 128);
  sphere(20);
  popMatrix();
  
  // transparent first, behind
  // opaque second in front
  pushMatrix();
  translate(width/1.5, height/4, 0);
  fill(0, 255, 0, 128);
  sphere(20);
  popMatrix();
  
  pushMatrix();
  translate(width/1.5, height/4, 30);
  fill(255);
  sphere(20);
  popMatrix();

  // opaque first in front
  // transparent second, behind  
  pushMatrix();
  translate(width/4, height/1.5, 30);
  fill(255);
  sphere(20);
  popMatrix();
  
  pushMatrix();
  translate(width/4, height/1.5, 0);
  fill(0, 0, 256, 128);
  sphere(20);
  popMatrix();

  // transparent first, behind
  // opaque second in front
  pushMatrix();
  translate(width/1.5, height/1.5, 30);
  fill(255, 255, 0, 128);
  sphere(20);
  popMatrix();
  
  pushMatrix();
  translate(width/1.5, height/1.5, 0);
  fill(255);
  sphere(20);
  popMatrix();
}
